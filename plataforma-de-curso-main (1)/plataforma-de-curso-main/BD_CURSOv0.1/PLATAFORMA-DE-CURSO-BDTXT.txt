CREATE TABLE CURSO(
ID_CURSO INT AUTO_INCREMENT PRIMARY KEY,
NOME_CURSO VARCHAR(70) NOT NULL,
DESC_CURSO TEXT NOT NULL,
ID_USUARIO_FK INT NOT NULL,
FOREIGN KEY(ID_USUARIO_FK) REFERENCES USUARIO(ID_USUARIO)
);

CREATE TABLE MODULO(
ID_MODULO INT AUTO_INCREMENT PRIMARY KEY,
NOME_MODULO VARCHAR(70) NOT NULL,
DESC_MODULO TEXT NOT NULL,
ID_CURSO_FK INT NOT NULL,
FOREIGN KEY (ID_CURSO_FK) REFERENCES CURSO(ID_CURSO)
);

CREATE TABLE AULA(
ID_AULA INT AUTO_INCREMENT PRIMARY KEY,
TITLE_AULA VARCHAR(70) NOT NULL,
CONTENT_AULA VARCHAR(280) NOT NULL,
ID_MODULO_FK INT NOT NULL,
FOREIGN KEY (ID_MODULO_FK) REFERENCES MODULO (ID_MODULO)
);

CREATE TABLE PARTICIPANTES(
ID_PARTICIPANTES INT AUTO INCREMENT PRIMARY KEY,
ID_USUARIOS_FK INT NOT NULL,
FOREIGN KEY(ID_USUARIOS_FK) REFERENCES USUARIO(ID_USUARIO),
ID_CURSO_FK NOT NULL,
FOREIGN KEY(ID_CURSO_FK) REFERENCES CURSO(ID_CURSO),
PARTY_TYPE VARCHAR(20) NOT NULL
);
